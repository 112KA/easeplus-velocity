Custom Back

  • customBackOut

    // イージング関数を新たに登録する、名前は任意、重複してたら上書き
// デフォルトは1.7
$.easeplus.addBackOut("customBackOut", 4.0);
$element.velocity({ translateX:300 }, 400, "customBackOut");
...
// 使わなくなったら削除も可能
$.easeplus.removeBackOut("customBackOut");
